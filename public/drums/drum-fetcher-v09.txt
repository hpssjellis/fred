<script>

let myAllPatterns = []

let myGeneratedText = ''

const myApiUrls = {

bardet:
'https://raw.githubusercontent.com/gvellut/dmp_midi/master/input/Patterns_200.json',

drumkits:
'https://raw.githubusercontent.com/jakesgordon/javascript-drum-machine/master/data/patterns.json',

grooves:
'https://raw.githubusercontent.com/mledoze/countries/master/countries.json',

stepseq:
'https://raw.githubusercontent.com/nbrosowsky/tonejs-instruments/master/samples.json',

electro:
'https://raw.githubusercontent.com/teropa/drumkit/master/patterns.json',

midi:
'https://raw.githubusercontent.com/Tonejs/Midi/master/examples/midi.json'

}

const myInstrumentMap = {

"kick":"mySound0",
"bassdrum":"mySound0",

"lowTom":"mySound1",

"midTom":"mySound2",

"highTom":"mySound3",

"snare":"mySound4",
"rimshot":"mySound4",
"clap":"mySound4",

"closedHihat":"mySound5",
"openhihat":"mySound5",
"hihat":"mySound5",

"crash":"mySound6",

"ride":"mySound7"

}

const myDrumInfo = [

{id:"mySound0",name:"Bass Drum",color:"#ff2222"},
{id:"mySound1",name:"Low Tom",color:"#ff8800"},
{id:"mySound2",name:"Mid Tom",color:"#ffaa00"},
{id:"mySound3",name:"High Tom",color:"#ffff00"},
{id:"mySound4",name:"Snare",color:"#00ff00"},
{id:"mySound5",name:"HiHat",color:"#00ffff"},
{id:"mySound6",name:"Crash",color:"#0088ff"},
{id:"mySound7",name:"Ride",color:"#aa00ff"}

]

function myApiChanged(){

    let myApi =
    document.getElementById('myApiSelect').value

    if(myApi === 'custom'){

        return

    }

    document.getElementById('myCustomUrl').value =
    myApiUrls[myApi]

}

window.onload = function(){

    myApiChanged()

}

async function myLoadPatterns(){

    let myApi =
    document.getElementById('myApiSelect').value

    let myUrl = ''

    if(myApi === 'custom'){

        myUrl =
        document.getElementById('myCustomUrl').value

    } else {

        myUrl = myApiUrls[myApi]

    }

    document.getElementById('myOutput').value =
    'Loading:\\n\\n' + myUrl

    try{

        let myResponse =
        await fetch(myUrl)

        if(!myResponse.ok){

            throw new Error(
            'Fetch Error: ' + myResponse.status
            )

        }

        let myData =
        await myResponse.json()

        console.log(myData)

        myAllPatterns =
        myNormalizePatterns(myData)

        myPopulatePatternSelect()

        document.getElementById('myOutput').value =
        'SUCCESS\\n\\nLoaded ' +
        myAllPatterns.length +
        ' patterns'

    } catch(myError){

        console.log(myError)

        document.getElementById('myOutput').value =
        'ERROR\\n\\n' +
        myError

    }

}

function myNormalizePatterns(myData){

    // Bardet format

    if(myData.patterns){

        return myParseBardet(myData)

    }

    // Generic array

    if(Array.isArray(myData)){

        return myParseArray(myData)

    }

    // ToneJS midi example

    if(myData.tracks){

        return myParseMidi(myData)

    }

    return []

}

function myParseBardet(myData){

    let myResult = []

    myData.patterns.forEach((myPattern,myIndex)=>{

        myResult.push({

            name:
            myPattern.name ||
            ('Pattern ' + myIndex),

            notes:
            myPattern.notes || []

        })

    })

    return myResult

}

function myParseArray(myData){

    let myResult = []

    myData.forEach((myItem,myIndex)=>{

        let myNotes = []

        if(myItem.notes){

            myNotes = myItem.notes

        }

        myResult.push({

            name:
            myItem.name ||
            ('Pattern ' + myIndex),

            notes:
            myNotes

        })

    })

    return myResult

}

function myParseMidi(myData){

    let myResult = []

    myData.tracks.forEach((myTrack,myTrackIndex)=>{

        let myNotes = []

        if(myTrack.notes){

            myTrack.notes.forEach(myNote=>{

                let myStep =
                Math.floor(myNote.time * 4)

                let myInstrument =
                'kick'

                if(myNote.midi === 38){

                    myInstrument = 'snare'

                }

                if(myNote.midi === 42){

                    myInstrument = 'closedHihat'

                }

                myNotes.push({

                    instrument:myInstrument,
                    step:myStep

                })

            })

        }

        myResult.push({

            name:
            myTrack.name ||
            ('Midi Track ' + myTrackIndex),

            notes:myNotes

        })

    })

    return myResult

}

function myPopulatePatternSelect(){

    let mySelect =
    document.getElementById('myPatternSelect')

    mySelect.innerHTML = ''

    myAllPatterns.forEach((myPattern,myIndex)=>{

        let myOption =
        document.createElement('option')

        myOption.value = myIndex

        myOption.textContent =
        myIndex +
        ' : ' +
        myPattern.name

        mySelect.appendChild(myOption)

    })

}

function myGenerateConfig(){

    if(myAllPatterns.length === 0){

        alert('No patterns loaded')

        return

    }

    let myIndex =
    parseInt(
    document.getElementById('myPatternSelect').value
    )

    let myPattern =
    myAllPatterns[myIndex]

    let myGrid = {}

    myDrumInfo.forEach(myDrum=>{

        myGrid[myDrum.id] =
        new Array(16).fill(0)

    })

    myPattern.notes.forEach(myNote=>{

        let myInstrument =
        myNote.instrument

        if(!myInstrument){

            return

        }

        myInstrument =
        myInstrument.toLowerCase()

        let myTarget =
        myInstrumentMap[myInstrument]

        let myStep =
        parseInt(myNote.step)

        if(
            myTarget &&
            myStep >= 0 &&
            myStep < 16
        ){

            myGrid[myTarget][myStep] = 1

        }

    })

    let myText = ''

    myText +=
    '# Drum Config File\n'

    myText +=
    '# Generated Pattern\n\n'

    myText +=
    'Name : ' +
    myPattern.name +
    '\n'

    myText +=
    'BPM : ' +
    document.getElementById('myBpm').value +
    '\n'

    myText +=
    'Steps : 16\n\n'

    myDrumInfo.forEach(myDrum=>{

        let myLine = ''

        for(let myI=0; myI<16; myI++){

            if(myI > 0 && myI % 4 === 0){

                myLine += ' '

            }

            myLine +=
            myGrid[myDrum.id][myI]

        }

        myText +=

        myDrum.id +
        ' | ' +

        myDrum.name +
        ' | ' +

        myDrum.color +
        ' | ' +

        myLine +
        '\n'

    })

    myGeneratedText = myText

    document.getElementById('myOutput').value =
    myText

}

function mySaveTextfile(){

    if(myGeneratedText === ''){

        alert('Generate output first')

        return

    }

    let myBlob =
    new Blob(
    [myGeneratedText],
    {type:'text/plain'}
    )

    let myLink =
    document.createElement('a')

    myLink.href =
    URL.createObjectURL(myBlob)

    myLink.download =
    'drum-pattern.txt'

    document.body.appendChild(myLink)

    myLink.click()

    document.body.removeChild(myLink)

}
